STEP 1 — CREATE: desktop_app.py (NEW FILE)
════════════════════════════════════════════════════════════════════════

Create file: desktop_app.py in project root

```python
import webview
import threading
import time
import sys
import os
from pathlib import Path

# Fix path for PyInstaller .exe
if getattr(sys, 'frozen', False):
    BASE_DIR = Path(sys._MEIPASS)
    WORK_DIR = Path(sys.executable).parent
else:
    BASE_DIR = Path(__file__).parent
    WORK_DIR = BASE_DIR

# Change working directory to where aury.db and downloads/ should live
os.chdir(WORK_DIR)

# Import Flask app AFTER chdir
from web_server import app as flask_app

PORT = 5000

def start_flask():
    """Start Flask server in background thread."""
    flask_app.run(
        host='127.0.0.1',  # localhost only — desktop app security
        port=PORT,
        debug=False,
        use_reloader=False,
        threaded=True
    )

def wait_for_server():
    """Wait until Flask is ready."""
    import urllib.request
    for _ in range(20):  # try 20 times, 0.5s each = 10s max
        try:
            urllib.request.urlopen(f'http://127.0.0.1:{PORT}/')
            return True
        except:
            time.sleep(0.5)
    return False

def main():
    # Start Flask in background
    flask_thread = threading.Thread(target=start_flask, daemon=True)
    flask_thread.start()

    # Wait for server to be ready
    print("Starting AURY...")
    if not wait_for_server():
        print("Server failed to start!")
        sys.exit(1)

    # Create desktop window using pywebview
    window = webview.create_window(
        title='AURY — Smart Media Downloader',
        url=f'http://127.0.0.1:{PORT}/',
        width=1280,
        height=800,
        min_size=(900, 600),
        resizable=True,
        text_select=True,
        confirm_close=True,    # "Are you sure?" on close
        background_color='#1a1614',  # matches dark theme bg
    )

    # Window events
    def on_closed():
        print("AURY closed.")
        # Clean shutdown — stop any active downloads gracefully

    window.events.closed += on_closed

    # Start the window (blocking call)
    webview.start(
        debug=False,           # set True during development
        private_mode=False,    # allow cookies/session
        storage_path=str(WORK_DIR / '.webview_data')  # persistent storage
    )

if __name__ == '__main__':
    main()
```

  --add-data "static;static" ^

  --hidden-import "webview.platforms.winforms" ^
  --hidden-import "clr" ^
  --hidden-import "flask" ^

echo ================================================

STEP 5 — UPDATE: core/config.py (Download path for Windows)
════════════════════════════════════════════════════════════════════════

In core/config.py, update the DOWNLOAD_DIR default:

```python
import sys
import os
from pathlib import Path

# Detect if running as desktop app or web server
IS_DESKTOP = getattr(sys, 'frozen', False) or \
             os.environ.get('AURY_DESKTOP', '') == '1'

# Default download path
if IS_DESKTOP and sys.platform == 'win32':
    # Windows desktop: default to Google Drive path
    GOOGLE_DRIVE_PATH = Path(os.environ.get('USERPROFILE', 'C:/Users')) / \
        'Google Drive Streaming' / 'My Drive' / 'cllg' / 'AURY' / 'downloads'
    
    # Use Google Drive path if it exists, else use local Downloads
    if GOOGLE_DRIVE_PATH.exists():
        DOWNLOAD_DIR = GOOGLE_DRIVE_PATH
    else:
        DOWNLOAD_DIR = Path(os.environ.get('USERPROFILE', '.')) / \
                       'Downloads' / 'AURY'
else:
    # Replit / Web server / Linux: use local ./downloads
    DOWNLOAD_DIR = BASE_DIR / 'downloads'
```
